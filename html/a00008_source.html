00224     // must have lengths equal to or greater than one.
00225     NSString *s_1 = [singular substringToIndex:1];
00226     NSString *p_1 = [plural substringToIndex:1];
00227     NSString *s1_ = [singular substringFromIndex:1];
00228     NSString *p1_ = [plural substringFromIndex:1];
00229     if ([[s_1 uppercaseString] isEqualToString:[p_1 uppercaseString]])
00230     {
00231         [self addPluralRegularExpressionRule:[NSString stringWithFormat:@"(%@)%@$", s_1, s1_] options:NSRegularExpressionCaseInsensitive replacement:[@"$1" stringByAppendingString:p1_]];
00232         [self addPluralRegularExpressionRule:[NSString stringWithFormat:@"(%@)%@$", p_1, p1_] options:NSRegularExpressionCaseInsensitive replacement:[@"$1" stringByAppendingString:p1_]];

00236     {
00237         [self addPluralRegularExpressionRule:[NSString stringWithFormat:@"%@(?i)%@$", [s_1 uppercaseString], s1_] options:NSRegularExpressionCaseInsensitive replacement:[[p_1 uppercaseString] stringByAppendingString:p1_]];

00304 NSString *ASInflectorApplyRulesAndReplacements(NSArray *rulesAndReplacements, NSString *word)
00305 {
00306     NSMutableString *result = [[word mutableCopy] autorelease];
00307     // Plurals is an array of arrays. The elements have two or three
00308     // sub-elements. If two elements, the first specifies a regular expression
00309     // and the second its replacement template. If three elements, the first
00310     // specifies a string target, the second its replacement and the third its
00311     // string compare options.
00312     //
00313     // Why is this necessary? Simply because Objective-C has no polymorphic
00314     // “gsub” method which accepts either a regular expression or string. Nor
00315     // can string-based search and replacements incorporate any flags;
00316     // comparison options amount to extra baggage. Hence the array of plurals
00317     // polymorphically carries different kinds of rules and their
00318     // replacements. The number of elements, the array count, determines
00319     // behaviour.
00320     for (NSArray *ruleAndReplacement in rulesAndReplacements)
00321     {
00322         if ([ruleAndReplacement count] == 2)
00323         {
00324             NSRegularExpression *re = [ruleAndReplacement objectAtIndex:0];
00325             NSString *replacement = [ruleAndReplacement objectAtIndex:1];
00326             if ([re replaceMatchesInString:result options:0 range:NSMakeRange(0, [result length]) withTemplate:replacement])
00327             {
00328                 break;
00329             }
00330         }
00331         else if ([ruleAndReplacement count] == 3)
00332         {
00333             NSString *target = [ruleAndReplacement objectAtIndex:0];
00334             NSString *replacement = [ruleAndReplacement objectAtIndex:1];
00335             NSUInteger options = [[ruleAndReplacement objectAtIndex:2] unsignedIntegerValue];
00336             if ([result replaceOccurrencesOfString:target withString:replacement options:options range:NSMakeRange(0, [result length])])
00337             {
00338                 break;
00339             }
00340         }
00341     }
00342     return [[result copy] autorelease];
00343 }
