00053 NSArray *ASAllObjectsInJSONObject(id object)
00054 {
00055     // Importantly, do not use an enumerator to iterate the array elements. The
00056     // iterator will modify the array; the count of objects increases at
00057     // iterations where the object is a non-empty array or dictionary. The array
00058     // has a two-fold function. First, to unwrap the arrays and
00059     // dictionaries. Second, to collect the array and dictionary contents for
00060     // checking; JSON objects should never contain circular references.
00061     NSMutableArray *objects = [NSMutableArray arrayWithObject:object];
00062     for (NSUInteger index = 0; index < [objects count]; index++)
00063     {
00064         id object = [objects objectAtIndex:index];
00065         if ([object isKindOfClass:[NSArray class]])
00066         {
00067             if ([objects firstObjectCommonWithArray:(NSArray *)object])
00068             {
00069                 return nil;
00070             }
00071             [objects addObjectsFromArray:(NSArray *)object];
00072         }
00073         else if ([object isKindOfClass:[NSDictionary class]])
00074         {
00075             NSArray *values = [(NSDictionary *)object allValues];
00076             if ([objects firstObjectCommonWithArray:values])
00077             {
00078                 return nil;
00079             }
00080             [objects addObjectsFromArray:values];
00081         }
00082     }
00083     return [objects copy];
00084 }
